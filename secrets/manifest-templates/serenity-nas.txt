apiVersion: networking.k8s.io/v1beta1
kind: Ingress
metadata:
  name: serenity-nas
  annotations:
    kubernetes.io/ingress.class: internal
    nginx.ingress.kubernetes.io/backend-protocol: "HTTP"
spec:
  rules:
  - host: "serenity.${DOMAIN}"
    http:
      paths:
      - backend:
          serviceName: serenity-nas
          servicePort: 8080
  tls:
  - hosts:
    - "serenity.${DOMAIN}"
---
apiVersion: v1
kind: Service
metadata:
  name: serenity-nas
spec:
  ports:
  - name: http
    port: 8080
  type: ExternalName
  externalName: 192.168.1.40
