apiVersion: networking.k8s.io/v1beta1
kind: Ingress
metadata:
  name: rclone-web
  annotations:
    kubernetes.io/ingress.class: internal
spec:
  rules:
  - host: "rclone.${DOMAIN}"
    http:
      paths:
      - backend:
          serviceName: rclone-web
          servicePort: http
  tls:
  - hosts:
    - "rclone.${DOMAIN}"
